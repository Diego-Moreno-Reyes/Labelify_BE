CREATE TABLE roles (
    id SERIAL PRIMARY KEY,
    name VARCHAR(50) NOT NULL UNIQUE,
    status BOOLEAN DEFAULT TRUE
);

CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(100) NOT NULL,
    password_hash TEXT NOT NULL,
    role_id INT NOT NULL,
    status BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    CONSTRAINT fk_users_roles
        FOREIGN KEY (role_id)
        REFERENCES roles(id)
        ON DELETE RESTRICT
);



CREATE TABLE printers (
    id SERIAL PRIMARY KEY,
    ip_address INET NOT NULL,
    port INT NOT NULL,
    status BOOLEAN NOT NULL
);

CREATE TABLE labels (
    id SERIAL PRIMARY KEY,
    
    part_number_dpo VARCHAR(50),
    part_number VARCHAR(50),

    vpps VARCHAR(50) NOT NULL,
    location_code VARCHAR(10) NOT NULL, 
    duns VARCHAR(50) NOT NULL,
    
    printing_limit INT NOT NULL, 
    printer_id INT NOT NULL, 

    CONSTRAINT fk_labels_printer
        FOREIGN KEY (printer_id) REFERENCES printers(id)
);

CREATE TABLE print_logs (
    id SERIAL PRIMARY KEY,

    label_id INT NOT NULL,

    serial_number VARCHAR(100) NOT NULL, 

    printed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    printed_by VARCHAR(100),
    
    reprint BOOLEAN NOT NULL, 

    CONSTRAINT fk_print_label
        FOREIGN KEY (label_id) REFERENCES labels(id)
);
